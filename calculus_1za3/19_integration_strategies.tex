    \begin{enumerate}
        \item Simplify the integrand if possible
            \begin{align*}
                \int (sin x + cocs x)^2 \ dx \\
                = \int (sin^2 x + 2 sin x cos x + cos^2 x) \ dx \\
                = \int 1 + sin 2x \ dx
            \end{align*}
        \item Look for obvious substitution
            \begin{align*}
                \int x^3 e^{x^4} \ dx \\
                u = x^4 \\
                \frac{1}{4} \int e^u \ du
            \end{align*}
        \item Classify the integrand according to its form and use a
            corresponding method
            \begin{enumerate}
                \item $\int \sin^m x \cos^n x \ dx$
                \item $\int \tan^m x \sec^n x \ dx$
                \item $\int \frac{P(x)}{Q(x)} \ dx$
            \end{enumerate}
        \item Integration by parts
        \item Use trig substitution for radicals
        \item Try another method
    \end{enumerate}
